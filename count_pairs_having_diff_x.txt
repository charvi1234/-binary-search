int comp1(const void *a,const void *b)
{
    return *(int*)a-*(int*)b;
}
int pairs(int k, int arr_count, int* arr) 
{
    qsort(arr,arr_count,sizeof(int),comp1);
int r=1,pair=0;
    int l=0;
    while(r<arr_count)
    {
        if(arr[r]-arr[l]==k)
        {
            pair=pair+1;
            r++;
        }
        else if(arr[r]-arr[l]>k)
            l=l+1;
        else
            r++;
    }
    return pair;

}
